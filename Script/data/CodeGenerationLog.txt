function daysInAdvanceDiscountLevel(ticketDate) {
  today = new Date()
  msPerDay = 24 * 60 * 60 * 1000
  DIA = (ticketDate.getTime() - today.getTime()) / msPerDay + 1
  DIA = Math.floor(DIA)

  return DIA 
}
